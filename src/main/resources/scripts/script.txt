-- Creación de tablespaces
CREATE TABLESPACE parametricas DATAFILE 'parametricas.dbf' SIZE 20M;
CREATE TABLESPACE transacciones DATAFILE 'transacciones.dbf' SIZE 20M;
CREATE TABLESPACE entidades DATAFILE 'entidades.dbf' SIZE 20M;
CREATE TABLESPACE indices DATAFILE 'indices.dbf' SIZE 20M;

CREATE TABLE cliente (
    cliente_id       NUMBER(10) NOT NULL,
    nombre           VARCHAR2(50) NOT NULL,
    apellido         VARCHAR2(50),
    direccion        VARCHAR2(100),
    email            VARCHAR2(100),
    telefono         VARCHAR2(15),
    fecha_nacimiento DATE,
    estado           VARCHAR2(8) NOT NULL
);

ALTER TABLE cliente ADD CONSTRAINT cliente_pk PRIMARY KEY ( cliente_id );

CREATE TABLE comision_jerarquica (
    nivel_id   NUMBER(10) NOT NULL,
    porcentaje NUMBER(5, 2)
);

ALTER TABLE comision_jerarquica ADD CONSTRAINT comision_jerarquica_pk PRIMARY KEY ( nivel_id );

CREATE TABLE cuenta (
    usuario        VARCHAR2(20) NOT NULL,
    contrasenia    VARCHAR2(30) NOT NULL,
    fecha_creacion DATE,
    persona_id     NUMBER(10) NOT NULL,
    estado         VARCHAR2(8) NOT NULL,
    tipo_usuario   VARCHAR2(10)
);

ALTER TABLE cuenta
    ADD CHECK ( tipo_usuario IN ( 'admin', 'cliente', 'vendedor' ) );

ALTER TABLE cuenta ADD CONSTRAINT cuenta_pk PRIMARY KEY ( usuario );

CREATE TABLE detalle_pedido (
    pedido_id NUMBER(13) NOT NULL,
    libro_id  NUMBER(10) NOT NULL,
    cantidad  NUMBER(3),
    subtotal  NUMBER(8, 2)
);

CREATE TABLE detalle_venta (
    venta_id    NUMBER(13) NOT NULL,
    cliente_id  NUMBER(10) NOT NULL,
    vendedor_id NUMBER(10) NOT NULL,
    libro_id    NUMBER(10) NOT NULL,
    cantidad    NUMBER(3),
    subtotal    NUMBER(8, 2)
);

CREATE TABLE inventario (
    libro_id   NUMBER(10) NOT NULL,
    existencia NUMBER(4)
);

ALTER TABLE inventario ADD CONSTRAINT inventario_pk PRIMARY KEY ( libro_id );

CREATE TABLE libro (
    libro_id     NUMBER(10) NOT NULL,
    titulo       VARCHAR2(100) NOT NULL,
    autor        VARCHAR2(50),
    precioventa  NUMBER(8, 2) NOT NULL,
    categoria    VARCHAR2(50),
    preciocompra NUMBER(8, 2) NOT NULL
);

ALTER TABLE libro ADD CONSTRAINT libro_pk PRIMARY KEY ( libro_id );

CREATE TABLE metodo_pago (
    metodo_pago_id NUMBER(10) NOT NULL,
    numero_tarjeta VARCHAR2(19),
    contrasenia    VARCHAR2(19),
    persona_id     NUMBER(10) NOT NULL
);

ALTER TABLE metodo_pago ADD CONSTRAINT metodo_pago_pk PRIMARY KEY ( metodo_pago_id );

CREATE TABLE nivel (
    nombre         VARCHAR2(20) NOT NULL,
    comision       NUMBER(5, 2) NOT NULL,
    descuento      NUMBER(5, 2) NOT NULL,
    ventas_minimas NUMBER(3)
);

ALTER TABLE nivel ADD CHECK ( descuento BETWEEN 0 AND 100 );

ALTER TABLE nivel ADD CONSTRAINT nivel_pk PRIMARY KEY ( nombre );

CREATE TABLE pago (
    pago_id        NUMBER(15),
    cantidad       NUMBER(10, 2),
    metodo_pago_id NUMBER(10) NOT NULL,
    venta_id       NUMBER(13) NOT NULL
);

CREATE TABLE pedido (
    pedido_id     NUMBER(13) NOT NULL,
    fecha_pedido  DATE,
    fecha_entrega DATE,
    proveedor_id  NUMBER(10) NOT NULL,
    estado        VARCHAR2(20) NOT NULL,
    total         NUMBER(10, 2)
);

ALTER TABLE pedido ADD CONSTRAINT pedido_pk PRIMARY KEY ( pedido_id );

CREATE TABLE proveedor (
    proveedor_id NUMBER(10) NOT NULL,
    nombre       VARCHAR2(100) NOT NULL,
    telefono     VARCHAR2(15),
    email        VARCHAR2(100),
    direccion    VARCHAR2(100)
);

ALTER TABLE proveedor ADD CONSTRAINT proveedor_pk PRIMARY KEY ( proveedor_id );

CREATE TABLE sueldo_vendedores (
    sueldo_id      NUMBER NOT NULL,
    vendedor_id    NUMBER(10) NOT NULL,
    sueldo_base    NUMBER(10, 2),
    comisiones     NUMBER(10, 2),
    fecha_pago     DATE,
    nivel          VARCHAR2(20),
    posicion_arbol VARCHAR2(100),
    estado         VARCHAR2(20) DEFAULT 'pendiente'
);

ALTER TABLE sueldo_vendedores ADD CONSTRAINT sueldo_vendedores_pk PRIMARY KEY ( sueldo_id );

CREATE TABLE vendedor (
    vendedor_id      NUMBER(10) NOT NULL,
    nombre           VARCHAR2(50) NOT NULL,
    fecha_nacimiento DATE,
    jefe_id          NUMBER(10),
    nivel            VARCHAR2(50) NOT NULL,
    estado           VARCHAR2(8) NOT NULL
);

ALTER TABLE vendedor ADD CONSTRAINT vendedor_pk PRIMARY KEY ( vendedor_id );

CREATE TABLE venta (
    venta_id NUMBER(13) NOT NULL,
    fecha    DATE,
    estado   VARCHAR2(20) NOT NULL,
    total    NUMBER(10, 2)
);

ALTER TABLE venta ADD CONSTRAINT venta_pk PRIMARY KEY ( venta_id );

CREATE SEQUENCE metodo_pago_metodo_pago_id_seq START WITH 1 NOCACHE ORDER;

CREATE OR REPLACE TRIGGER metodo_pago_metodo_pago_id_trg BEFORE
    INSERT ON metodo_pago
    FOR EACH ROW
    WHEN ( new.metodo_pago_id IS NULL )
BEGIN
    :new.metodo_pago_id := metodo_pago_metodo_pago_id_seq.nextval;
END;
/

CREATE SEQUENCE pago_pago_id_seq START WITH 1 NOCACHE ORDER;

CREATE OR REPLACE TRIGGER pago_pago_id_trg BEFORE
    INSERT ON pago
    FOR EACH ROW
    WHEN ( new.pago_id IS NULL )
BEGIN
    :new.pago_id := pago_pago_id_seq.nextval;
END;
/

CREATE SEQUENCE sueldo_vendedores_sueldo_id START WITH 1 NOCACHE ORDER;

CREATE OR REPLACE TRIGGER sueldo_vendedores_sueldo_id BEFORE
    INSERT ON sueldo_vendedores
    FOR EACH ROW
    WHEN ( new.sueldo_id IS NULL )
BEGIN
    :new.sueldo_id := sueldo_vendedores_sueldo_id.nextval;

/

CREATE SEQUENCE venta_venta_id_seq START WITH 1 NOCACHE ORDER;

CREATE OR REPLACE TRIGGER venta_venta_id_trg BEFORE
    INSERT ON venta
    FOR EACH ROW
    WHEN ( new.venta_id IS NULL )

    :new.venta_id := venta_venta_id_seq.nextval;

CREATE SEQUENCE seq_pedido_id START WITH 1 NOCACHE ORDER;

CREATE OR REPLACE TRIGGER pedido_pedido_id_trg BEFORE
	INSERT ON Pedido
	FOR EACH ROW
	WHEN (new.pedido_id IS NULL)

-- Creación de roles
CREATE ROLE vendedor;
CREATE ROLE cliente;
CREATE ROLE administrador;

-- Asignación de permisos para roles
-- Permisos para el rol cliente
GRANT SELECT, INSERT, UPDATE ON cliente TO cliente;
GRANT SELECT, INSERT ON cuenta TO cliente;
GRANT SELECT ON inventario TO cliente;
GRANT SELECT, INSERT ON metodo_pago TO cliente;
GRANT UPDATE (NUMERO_TARJETA, CONTRASENIA) on METODO_PAGO TO CLIENTE;

-- Permisos para el rol vendedor
GRANT SELECT, INSERT, UPDATE ON vendedor TO vendedor;
GRANT SELECT, INSERT, UPDATE ON cuenta TO vendedor;
GRANT SELECT ON inventario TO vendedor;
GRANT SELECT, INSERT, UPDATE ON venta TO vendedor;
GRANT SELECT, INSERT ON detalle_venta TO vendedor;
GRANT SELECT ON nivel TO vendedor;
GRANT SELECT, INSERT, UPDATE ON sueldo_vendedores TO vendedor;

-- Permisos para el rol administrador
GRANT ALL PRIVILEGES ON cliente TO administrador;
GRANT ALL PRIVILEGES ON cuenta TO administrador;
GRANT ALL PRIVILEGES ON inventario TO administrador;
GRANT ALL PRIVILEGES ON metodo_pago TO administrador;
GRANT ALL PRIVILEGES ON nivel TO administrador;
GRANT ALL PRIVILEGES ON pago TO administrador;
GRANT ALL PRIVILEGES ON pedido TO administrador;
GRANT ALL PRIVILEGES ON proveedor TO administrador;
GRANT ALL PRIVILEGES ON sueldo_vendedores TO administrador;
GRANT ALL PRIVILEGES ON vendedor TO administrador;
GRANT ALL PRIVILEGES ON venta TO administrador;
GRANT ALL PRIVILEGES ON detalle_venta TO administrador;
GRANT ALL PRIVILEGES ON detalle_pedido TO administrador;
GRANT ALL PRIVILEGES ON libro TO administrador;
GRANT ALL PRIVILEGES ON comision_jerarquica TO administrador;

-- Creación de usuarios y asignación de roles
CREATE USER cliente_user IDENTIFIED BY cliente_pass DEFAULT TABLESPACE entidades;
CREATE USER vendedor_user IDENTIFIED BY vendedor_pass DEFAULT TABLESPACE entidades;
CREATE USER admin_user IDENTIFIED BY admin_pass DEFAULT TABLESPACE entidades;
GRANT cliente TO cliente_user;
GRANT vendedor TO vendedor_user;
GRANT administrador TO admin_user;
